Scripting for presentation release

# In local working copy of bssw-tutorial/presentations repository
# Ensure destination directory exists and copy/move files there
mkdir -p final-presentations/2022-08-11-atpesc
cp presentations.yml final-presentations/2022-08-11-atpesc
mv intro-a.pdf final-presentations/2022-08-11-atpesc/01-intro-a.pdf
mv overview-a.pdf final-presentations/2022-08-11-atpesc/02-overview-a.pdf
mv design-a.pdf final-presentations/2022-08-11-atpesc/03-design-a.pdf
mv licensing.pdf final-presentations/2022-08-11-atpesc/04-licensing.pdf
mv spack.pdf final-presentations/2022-08-11-atpesc/05-spack.pdf
mv testing-a.pdf final-presentations/2022-08-11-atpesc/06-testing-a.pdf
mv lab-notebooks.pdf final-presentations/2022-08-11-atpesc/07-lab-notebooks.pdf
mv computational-expts.pdf final-presentations/2022-08-11-atpesc/08-computational-expts.pdf
mv mining-tools.pdf final-presentations/2022-08-11-atpesc/09-mining-tools.pdf

# Add and commit new files and tag repository
cd final-presentations/2022-08-11-atpesc
git add *.pdf presentations.yml
git commit -m "Capturing presentations for 2022-08-11-atpesc"
git push

# Create archive
zip --update 2022-08-11-atpesc.zip *.pdf

# Tag repo
git tag -a 2022-08-11-atpesc -m "2022-08-11: Software Productivity and Sustainability track @ Argonne Training Program on Extreme-Scale Computing summer school"
git push origin --tags

# Create release and add zip archive
gh release create 2022-08-11-atpesc --title "2022-08-11: Software Productivity and Sustainability track @ Argonne Training Program on Extreme-Scale Computing summer school" --notes "2022-08-11: Software Productivity and Sustainability track @ Argonne Training Program on Extreme-Scale Computing summer school"
gh release upload 2022-08-11-atpesc 2022-08-11-atpesc.zip

Scripting to remove tag and release

For when we have to update the presentations.

# In local working copy of bssw-tutorial/presentations repository
# Delete release (--yes overrides confirmation request)
gh release delete 2022-08-11-atpesc --yes

# Delete local and remote tags
git tag -d 2022-08-11-atpesc
git push --delete origin 2022-08-11-atpesc

# Start over at "tag repo" step above to create new release
