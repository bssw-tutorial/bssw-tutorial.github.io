Scripting for presentation release

# In local working copy of bssw-tutorial/presentations repository
# Ensure destination directory exists and copy/move files there
mkdir -p final-presentations/2023-04-18-iss
cp presentations.yml final-presentations/2023-04-18-iss
mv intro.pdf final-presentations/2023-04-18-iss/01-intro.pdf
mv overview.pdf final-presentations/2023-04-18-iss/02-overview.pdf
mv design.pdf final-presentations/2023-04-18-iss/03-design.pdf
mv refactoring.pdf final-presentations/2023-04-18-iss/04-refactoring.pdf
mv collaborative-sw-dev.pdf final-presentations/2023-04-18-iss/05-collaborative-sw-dev.pdf
mv packaging-short.pdf final-presentations/2023-04-18-iss/06-packaging-short.pdf
mv testing-advanced-short.pdf final-presentations/2023-04-18-iss/07-testing-advanced-short.pdf
mv reproducibility.pdf final-presentations/2023-04-18-iss/08-reproducibility.pdf
mv summary.pdf final-presentations/2023-04-18-iss/09-summary.pdf

# Add and commit new files and tag repository
cd final-presentations/2023-04-18-iss
git add *.pdf presentations.yml doi.txt
git commit -m "Capturing presentations for 2023-04-18-iss"
git push

# Create archive
zip --update 2023-04-18-iss.zip *.pdf

# Tag repo
git tag -a 2023-04-18-iss -m "2023-04-18: Better Scientific Software tutorial @ Improving Scientific Software conference"
git push origin --tags

# Create release and add zip archive
gh release create 2023-04-18-iss --title "2023-04-18: Better Scientific Software tutorial @ Improving Scientific Software conference" --notes "2023-04-18: Better Scientific Software tutorial @ Improving Scientific Software conference"
gh release upload 2023-04-18-iss 2023-04-18-iss.zip

Scripting to remove tag and release

For when we have to update the presentations.

# In local working copy of bssw-tutorial/presentations repository
# Delete release (--yes overrides confirmation request)
gh release delete 2023-04-18-iss --yes

# Delete local and remote tags
git tag -d 2023-04-18-iss
git push --delete origin 2023-04-18-iss

# Start over at "tag repo" step above to create new release
